1#!/usr/bin/env python3 

2"""Implements 'pam_script_auth' module for pam-script subsystem. 

3 

4The environment variables passed by pam-script onto the script 

5(all will exist but some may be null if not applicable): 

6 

7PAM_SERVICE - the application that's invoking the PAM stack 

8PAM_TYPE - the module-type (e.g. auth_data,account,session,password) 

9PAM_USER - the user being authenticated into 

10PAM_RUSER - the remote user, the user invoking the application 

11PAM_RHOST - remote host 

12PAM_TTY - the controlling tty 

13PAM_AUTHTOK - password in readable text 

14PAM_OLDAUTHTOK - old password in readable text 

15 

16Id addition the pam_script.so arguments in the pam.conf will be passed 

17on the command line, which can be used to modify the script behavior. 

18 

19dir - installation dir 

20grace - time 'skew' allowing the validation (in seconds) 

21check_timeframe - validates the SAML assertion if expired (True/False) 

22idp - trusted IdPs metadata file (CSV multi-value) 

23log_level - logging severity 

24only_from - trusted IdPs host names (CSV multi-value) 

25trusted_sp - entityID of trusted SP 

26user_id - Attribute element representing validated username 

27"""

175def decode_assertion(data): 

176 """Base64 decodes and then inflates SAML assertion data. 

177 

178 :param data: base64 encoded and deflated SAML assertion data (as 

179 received from IdP) 

180 :return: decoded and inflated SAML assertion string 

181 """ 

182 

183 data_decoded = b64decode(data) 

184 try: 

185 return zlib.decompress(data_decoded) 

186 except zlib.error: 

187 pass 

188 return data_decoded

205def parse_idp_metadata(idp_metadata_file): 

206 """Parse entityID and X509Certificate elements from metadata file. 

207 

208 :param idp_metadata_file: IdP metadata file name 

209 :return: dictionary of element values 

210 """ 

211 

212 logger = logging.getLogger(__pam_module_name__) 

213 data = {'entityID': '', 'x509cert': ''} 

214 

215 try: 

216 with open(idp_metadata_file, "rb") as fh: 

217 md_tree = etree.parse(fh).getroot() 

218 except (OSError, IOError) as err: 

219 logger.warning( 

220 f"IdP metadata file is missing or corrupted: {err}.") 

221 return data 

222 

223 data['entityID'] = dict(md_tree.items()).get('entityID') 

224 signing_nodes = md_tree.findall(".//md:KeyDescriptor[@use='signing']", 

225 namespaces=ns) 

226 

227 if len(signing_nodes) > 0: 

228 certs = [] 

229 for cert_node in signing_nodes: 

230 certs.append(''.join(cert_node.find( 

231 ".//ds:KeyInfo/ds:X509Data/ds:X509Certificate", 

232 namespaces=ns).text.split())) 

233 data['x509cert'] = certs 

234 

235 return data

238def iterate_certs(data): 

239 """Iterate over all (entityID, x509cert) pairs. 

240 

241 :param data: parsed IdP data 

242 :return: entityID, cert 

243 """ 

244 

245 for idp in data: 

246 eid = idp['entityID'] 

247 for crt in idp['x509cert']: 

248 yield eid, crt

251def get_timestamps(etree_xml): 

252 """Parse SAML assertion validity timestamps 

253 

254 Timestamps are parsed from Conditions element and converted to Unix 

255 timestamps. 

256 

257 :param etree_xml: etree element object 

258 :return: NotBefore, NotOnOrAfter 

259 """ 

260 

261 logger = logging.getLogger(__pam_module_name__) 

262 time_attr = {'NotBefore': 0, 'NotOnOrAfter': 0} 

263 

264 conditions_node = etree_xml.find(".//saml:Conditions", namespaces=ns) 

265 

266 if conditions_node is not None: 

267 time_attr['NotBefore'] = conditions_node.get('NotBefore') 

268 time_attr['NotOnOrAfter'] = conditions_node.get('NotOnOrAfter') 

269 

270 for key, timestamp in time_attr.items(): 

271 if timestamp is not None: 

272 try: 

273 dtime = ciso8601.parse_datetime(timestamp) 

274 time_attr[key] = calendar.timegm(dtime.utctimetuple()) 

275 except ValueError as err: 

276 logger.error( 

277 f"Time string {key}={timestamp} does not match " 

278 f"expected format: {err}.") 

279 sys.exit(PAM_CRED_ERR) 

280 

281 return [(0 if v is None else v) for v in time_attr.values()]

284def verify_timestamps(nb, nooa, grace): 

285 """Verify if given timeframe (including grace) is valid. 

286 

287 :param nb: 

288 :param nooa: 

289 :param grace: grace/skew time period in seconds 

290 :return: True/False 

291 """ 

292 

293 logger = logging.getLogger(__pam_module_name__) 

294 

295 now = int(time.time()) 

296 

297 logger.info( 

298 f"SAML assertion validity: NB={time.ctime(nb)} " 

299 f"NOOA={time.ctime(nooa)} (grace={grace}).") 

300 

301 if nb and nb > (now + grace): 

302 logger.error("SAML assertion timestamps verification " 

303 "failed: timestamps not yet valid.") 

304 return False 

305 if nooa and (nooa + grace) <= now: 

306 logger.error("SAML assertion timestamps verification " 

307 "failed: timestamps expired.") 

308 return False 

309 return True

312def verify_assertion_signature(auth_data, idp_metadata): 

313 """Verify SAML assertion signature 

314 

315 :param auth_data: encoded SAML assertion 

316 :param idp_metadata: IdP metadata 

317 :return: verified assertion tree element 

318 """ 

319 

320 logger = logging.getLogger(__pam_module_name__) 

321 

322 assertion = decode_assertion(auth_data) 

323 verified_assertion = b'' 

324 

325 # Extract IdPs signing certificates 

326 idp_data = [] 

327 for file in [file.strip() for file in idp_metadata.split(',')]: 

328 idp_data.append(parse_idp_metadata(file)) 

329 

330 for entity, cert in iterate_certs(idp_data): 

331 try: 

332 verified_assertion = XMLVerifier().verify( 

333 assertion, 

334 x509_cert=cert, 

335 validate_schema=True) 

336 logger.debug(f"SAML assertion signature verified for " 

337 f"IdP entityID={entity}.") 

338 break 

339 except (InvalidSignature, 

340 InvalidDigest, 

341 InvalidCertificate, 

342 InvalidInput) as assertion_verify_err: 

343 logger.warning( 

344 f"SAML assertion signature verification error for IdP " 

345 f"entityID={entity}: {assertion_verify_err}.") 

346 

347 if verified_assertion: 

348 verified_assertion = verified_assertion.signed_xml 

349 else: 

350 logger.error( 

351 "SAML assertion signatures can not be verified: " 

352 "no valid signature found.") 

353 sys.exit(PAM_AUTH_ERR) 

354 return verified_assertion

357def main(): 

358 """Implements 'pam_script_auth' module for pam-script subsystem. 

359 

360 :return: PAM compatible exit code 

361 """ 

362 

363 pam_params = get_pam_params(os.environ, sys.argv[1:]) 

364 logger = config_logging(pam_params.get('log_level')) 

365 

366 # Verify PAM_TYPE request 

367 pam_type = pam_params.get('PAM_TYPE') 

368 if pam_type != 'auth_data': 

369 logger.error(f"Unsupported PAM_TYPE={pam_type} requested.") 

370 sys.exit(PAM_MODULE_UNKNOWN) 

371 

372 # Verify 'only_from' response conditions 

373 only_from = pam_params.get('only_from') 

374 pam_rhost = pam_params.get('PAM_RHOST') 

375 

376 if only_from and pam_rhost and pam_rhost not in [host.strip() for host in 

377 only_from.split(',')]: 

378 logger.error( 

379 f"Requesting remote host PAM_RHOST={pam_rhost} is not allowed to " 

380 f"authenticate.") 

381 sys.exit(PAM_CRED_INSUFFICIENT) 

382 

383 # Verify SAML assertion signature 

384 tree_verified = verify_assertion_signature(pam_params.get('PAM_AUTHTOK'), 

385 pam_params.get('idp')) 

386 

387 # Verify trusted SP 

388 trusted_sp = pam_params['trusted_sp'] 

389 if not trusted_sp: 

390 logger.warning( 

391 "Unsecured configuration: no trusted_sp argument defined.") 

392 else: 

393 node = tree_verified.find( 

394 ".//saml:AudienceRestriction/saml:Audience", namespaces=ns) 

395 if node is not None and node.text and node.text == trusted_sp: 

396 logger.debug( 

397 f"SAML assertion element Audience " 

398 f"match trusted_sp={trusted_sp}.") 

399 else: 

400 logger.error( 

401 f"SAML assertion element Audience " 

402 f"do not match trusted_sp={trusted_sp}.") 

403 sys.exit(PAM_CRED_INSUFFICIENT) 

404 

405 # Verify uid matching 

406 user_id = pam_params.get('user_id') 

407 user = get_uid_attribute(tree_verified, user_id) 

408 if user is not None: 

409 user = user.text 

410 logger.debug("SAML assertion attribute element with user_id " 

411 "value extracted.") 

412 else: 

413 logger.error( 

414 f"SAML assertion did not contain attribute element with " 

415 f"user_id={user_id} value.") 

416 sys.exit(PAM_AUTH_ERR) 

417 

418 pam_user = pam_params.get('PAM_USER') 

419 

420 if user and (user == pam_user): 

421 logger.debug( 

422 f"SAML assertion attribute element user_id={user_id} value match " 

423 f"to PAM_USER={pam_user}.") 

424 else: 

425 logger.error( 

426 f"SAML assertion attribute element user_id={user_id} value do " 

427 f"not match to PAM_USER={pam_user}.") 

428 sys.exit(PAM_AUTH_ERR) 

429 

430 # Verify timestamps 

431 if pam_params['check_timeframe']: 

432 nb, nooa = get_timestamps(tree_verified) 

433 if verify_timestamps(nb, nooa, pam_params['grace']): 

434 logger.debug("SAML assertion timestamps verified.") 

435 else: 

436 logger.error("SAML assertion timestamps verification failed.") 

437 sys.exit(PAM_CRED_EXPIRED) 

438 else: 

439 logger.warning( 

440 "SAML assertion time validity not checked: disabled by " 

441 "check_timeframe=False parameter.") 

442 

443 logger.info(f"SAML assertion verified: user={user} allowed to login.") 

444 sys.exit(PAM_SUCCESS)

448 main()
